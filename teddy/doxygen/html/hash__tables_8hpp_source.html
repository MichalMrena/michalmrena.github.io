277inline auto table_base::get_gte_capacity(int64 const capacity) -> int64
278{
279 auto const predicate = [capacity] (int64 const currentCapacity)
280 {
281 return currentCapacity > capacity;
282 };
283 auto const tableIt
284 = std::find_if(begin(Capacities), end(Capacities), predicate);
285 return tableIt == std::end(Capacities) ? Capacities.back() : *tableIt;
286}
