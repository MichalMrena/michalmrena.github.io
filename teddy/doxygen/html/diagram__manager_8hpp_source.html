893 template<class Data, degree Degree, domain Domain>
894 template<std::output_iterator<teddy::uint_t> O>
895 auto diagram_manager<Data, Degree, Domain>::to_vector_g
896 (diagram_t const& d, O out) const -> void
897 {
898 if (this->get_var_count() == 0)
899 {
900 assert(d.get_root()->is_terminal());
901 *out++ = d.get_root()->get_value();
902 return;
903 }
904
905 auto vars = std::vector<uint_t>(this->get_var_count());
906 auto wasLast = false;
907 do
908 {
909 *out++ = this->evaluate(d, vars);
910
911 auto overflow = true;
912 auto level = nodes_.get_leaf_level();
913 while (level > 0 && overflow)
914 {
915 --level;
916 auto const index = nodes_.get_index(level);
917 ++vars[index];
918 overflow = vars[index] == nodes_.get_domain(index);
919 if (overflow)
920 {
921 vars[index] = 0;
922 }
923
924 wasLast = overflow && 0 == level;
925 }
926 }
927 while (not wasLast);
928 }

933 ( pla_file const& file
934 , fold_type const foldType ) -> second_t<Foo, std::vector<diagram_t>>
935 {
936 auto const product = [this](auto const& cube)
937 {
938 auto vs = std::vector<diagram_t>();
939 vs.reserve(cube.size());
940 for (auto i = 0u; i < cube.size(); ++i)
941 {
942 if (cube.get(i) == 1)
943 {
944 vs.emplace_back(this->variable(i));
945 }
946 else if (cube.get(i) == 0)
947 {
948 vs.emplace_back(this->variable_not(i));
949 }
950 }

954 auto const orFold = [this, foldType](auto& ds)
955 {
956 switch (foldType)
957 {
958 case fold_type::Left:
959 return this->left_fold<ops::OR>(ds);
960
961 case fold_type::Tree:
962 return this->tree_fold<ops::OR>(ds);
963
964 default:
965 assert(false);
966 return this->constant(0);
967 }
968 };

970 auto const& plaLines = file.get_lines();
971 auto const lineCount = file.line_count();
972 auto const functionCount = file.function_count();
973
974 // Create a diagram for each function.
975 auto functionDiagrams = std::vector<diagram_t>();
976 functionDiagrams.reserve(functionCount);
977 for (auto fi = 0u; fi < functionCount; ++fi)
978 {
979 // First create a diagram for each product.
980 auto products = std::vector<diagram_t>();
981 products.reserve(lineCount);
982 for (auto li = 0u; li < lineCount; ++li)
983 {
984 // We are doing SOP so we are only interested
985 // in functions with value 1.
986 if (plaLines[li].fVals.get(fi) == 1)
987 {
988 products.emplace_back(product(plaLines[li].cube));
989 }
990 }
991
992 // In this case we just have a constant function.
993 if (products.empty())
994 {
995 products.emplace_back(this->constant(0));
996 }
997

1007 auto diagram_manager<Data, Degree, Domain>::apply
1008 (diagram_t const& d1, diagram_t const& d2) -> diagram_t
1009 {
1010 auto const go = [this](auto&& go_, auto const l, auto const r)
1011 {
1012 auto const cached = nodes_.template cache_find<Op>(l, r);
1013 if (cached)
1014 {
1015 return cached;
1016 }
1017
1018 auto const lhsVal = node_value(l);
1019 auto const rhsVal = node_value(r);
1020 auto const opVal = Op()(lhsVal, rhsVal);
1021 auto u = static_cast<node_t*>(nullptr);
1022
1023 if (opVal != Nondetermined)
1024 {
1025 u = nodes_.terminal_node(opVal); // TODO osetrit pre specialne hodnoty
1026 }
1027 else
1028 {
1029 auto const lhsLevel = nodes_.get_level(l);
1030 auto const rhsLevel = nodes_.get_level(r);
1031 auto const topLevel = std::min(lhsLevel, rhsLevel);
1032 auto const topNode = topLevel == lhsLevel ? l : r;
1033 auto const topIndex = topNode->get_index();
1034 auto sons = nodes_.make_sons(topIndex, [=, &go_](auto const k)
1035 {
1036 auto const fst = lhsLevel == topLevel ? l->get_son(k) : l;
1037 auto const snd = rhsLevel == topLevel ? r->get_son(k) : r;
1038 return go_(go_, fst, snd);
1039 });
1040
1041 u = nodes_.internal_node(topIndex, std::move(sons));
1042 }
1043
1044 nodes_.template cache_put<Op>(l, r, u);
1045 return u;
1046 };
1047
1048 auto const r = go(go, d1.get_root(), d2.get_root());
1049 auto const d = diagram_t(r);
1050 return d;

1092 template<class Data, degree Degree, domain Domain>
1093 template< bin_op Op
1094 , std::random_access_iterator I
1095 , std::sentinel_for<I> S >
1096 auto diagram_manager<Data, Degree, Domain>::tree_fold
1097 (I first, S const last) -> diagram_t
1098 {
1099 static_assert(std::same_as<std::iter_value_t<I>, diagram_t>);
1100
1101 auto const count = std::distance(first, last);
1102 auto currentCount = count;
1103 auto const numOfSteps
1104 = static_cast<std::size_t>(std::ceil(std::log2(count)));
1105
1106 for (auto step = 0u; step < numOfSteps; ++step)
1107 {
1108 auto const justMoveLast = currentCount & 1;
1109 currentCount = (currentCount / 2) + justMoveLast;
1110 auto const pairCount = currentCount - justMoveLast;
1111
1112 for (auto i = 0u; i < pairCount; ++i)
1113 {
1114 *(first + i) = this->apply<Op>( *(first + 2 * i)
1115 , *(first + 2 * i + 1) );
1116 }
1117
1118 if (justMoveLast)
1119 {
1120 *(first + currentCount - 1)
1121 = std::move(*(first + 2 * (currentCount - 1)));
1122 }
1123 }
1124
1125 return diagram_t(std::move(*first));
1126 }

1153 template<class Data, degree Degree, domain Domain>
1154 auto diagram_manager<Data, Degree, Domain>::satisfy_count
1155 (uint_t const val, diagram_t& d) -> std::size_t
1156 {
1157 if constexpr (domains::is_fixed<Domain>()())
1158 {
1159 assert(val < Domain()());
1160 }
1161
1162 auto constexpr CanUseDataMember
1163 = std::is_floating_point_v<Data> or std::is_integral_v<Data>;
1164 using T = std::conditional_t<CanUseDataMember, Data, std::size_t>;
1165
1166 // A function that returns reference to
1167 // the data associated with given node.
1168 auto data = []()
1169 {
1170 if constexpr (CanUseDataMember)
1171 {
1172 // Simply return reference to the data member.
1173 return [](auto const n) mutable -> decltype(auto)
1174 {
1175 return (n->data());
1176 };
1177 }
1178 else
1179 {
1180 // Return reference to the data that is stored in the map.
1181 return [map = std::unordered_map<node_t*, T>()]
1182 (auto const n) mutable -> T&
1183 {
1184 // If there is no value for given key [] creates new pair
1185 // and value-initializes the value (0 for primitive types).
1186 return map[n];
1187 };
1188 }
1189 }();
1190
1191 // Actual satisfy count algorithm.
1192 nodes_.traverse_post(d.get_root(), [this, val, &data]
1193 (auto const n) mutable
1194 {
1195 if (n->is_terminal())
1196 {
1197 data(n) = n->get_value() == val ? 1 : 0;
1198 }
1199 else
1200 {
1201 data(n) = 0;
1202 auto const nLevel = nodes_.get_level(n);
1203 nodes_.for_each_son(n, [=, this, &data](auto const son) mutable
1204 {
1205 auto const sonLevel = nodes_.get_level(son);
1206 auto const diff = nodes_.domain_product( nLevel + 1
1207 , sonLevel );
1208 data(n) += data(son) * static_cast<T>(diff);
1209 });
1210 }
1211 });
1212
1213 auto const rootAlpha = static_cast<std::size_t>(data(d.get_root()));
1214 auto const rootLevel = nodes_.get_level(d.get_root());
1215 return rootAlpha * nodes_.domain_product(0, rootLevel);
1216 }

1322 if (n->is_terminal())

1338 = nodes_.internal_node(n->get_index(), std::move(sons));
1339 memo.emplace(n, newN);

1341 };
