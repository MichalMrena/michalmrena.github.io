851template<class Data, degree Degree, domain Domain>
852template<std::input_iterator I, std::sentinel_for<I> S>
853auto diagram_manager<Data, Degree, Domain>::from_vector(I first, S last)
854 -> diagram_t
855{
856 if (0 == this->get_var_count())
857 {
858 assert(first != last && std::next(first) == last);
859 return diagram_t(nodes_.make_terminal_node(*first));
860 }
861
862 auto const lastLevel = static_cast<int32>(this->get_var_count() - 1);
863 auto const lastIndex = nodes_.get_index(lastLevel);
864
865 if constexpr (std::random_access_iterator<I>)
866 {
867 [[maybe_unused]] auto const count
868 = nodes_.domain_product(0, lastLevel + 1);
869 [[maybe_unused]] auto const dist
870 = static_cast<int64>(std::distance(first, last));
871 assert(dist > 0 && dist == count);
872 }
873
874 using stack_frame = struct
875 {
876 node_t* node;
877 int32 level;
878 };
879
880 auto stack = std::vector<stack_frame>();
881 auto const shrink_stack = [this, &stack] ()
882 {
883 for (;;)
884 {
885 auto const currentLevel = stack.back().level;
886 if (0 == currentLevel)
887 {
888 break;
889 }
890
891 auto const endId = rend(stack);
892 auto stackIt = rbegin(stack);
893 auto count = 0;
894 while (stackIt != endId and stackIt->level == currentLevel)
895 {
896 ++stackIt;
897 ++count;
898 }
899 auto const newIndex = nodes_.get_index(currentLevel - 1);
900 auto const newDomain = nodes_.get_domain(newIndex);
901
902 if (count < newDomain)
903 {
904 break;
905 }
906
907 auto newSons = nodes_.make_sons(
908 newIndex,
909 [&stack, newDomain] (auto const sonOrder)
910 {
911 return stack[as_uindex(ssize(stack) - newDomain + sonOrder)]
912 .node;
913 }
914 );
915 auto const newNode
916 = nodes_.make_internal_node(newIndex, std::move(newSons));
917 stack.erase(end(stack) - newDomain, end(stack));
918 stack.push_back(stack_frame {newNode, currentLevel - 1});
919 }
920 };
921
922 while (first != last)
923 {
924 auto sons = nodes_.make_sons(
925 lastIndex,
926 [this, &first] (auto const)
927 {
928 return nodes_.make_terminal_node(*first++);
929 }
930 );
931 auto const node = nodes_.make_internal_node(lastIndex, std::move(sons));
932 stack.push_back(stack_frame {node, lastLevel});
933 shrink_stack();
934 }
935
936 assert(stack.size() == 1);
937 return diagram_t(stack.back().node);
938}

957template<class Data, degree Degree, domain Domain>
958template<std::output_iterator<teddy::int32> O>
959auto diagram_manager<Data, Degree, Domain>::to_vector_g(
960 diagram_t const& diagram,
961 O out
962) const -> void
963{
964 if (this->get_var_count() == 0)
965 {
966 assert(diagram.unsafe_get_root()->is_terminal());
967 *out++ = diagram.unsafe_get_root()->get_value();
968 return;
969 }
970
971 auto vars = std::vector<int32>(as_usize(this->get_var_count()));
972 auto wasLast = false;
973 do
974 {
975 *out++ = this->evaluate(diagram, vars);
976
977 auto overflow = true;
978 auto level = nodes_.get_leaf_level();
979 while (level > 0 && overflow)
980 {
981 --level;
982 auto const index = nodes_.get_index(level);
983 ++vars[as_uindex(index)];
984 overflow = vars[as_uindex(index)] == nodes_.get_domain(index);
985 if (overflow)
986 {
987 vars[as_uindex(index)] = 0;
988 }
989
990 wasLast = overflow && 0 == level;
991 }
992 } while (not wasLast);
993}

996template<class Foo>
997requires(is_bdd<Degree>)
998auto diagram_manager<Data, Degree, Domain>::from_pla(
999 pla_file const& file,
1000 fold_type const foldType
1001) -> second_t<Foo, std::vector<diagram_t>>
1002{
1003 auto const product = [this] (auto const& cube)
1004 {
1005 auto variables = std::vector<diagram_t>();
1006 variables.reserve(cube.size());
1007 for (auto i = 0; i < cube.size(); ++i)
1008 {
1009 if (cube.get(i) == 1)
1010 {
1011 variables.emplace_back(this->variable(i));
1012 }
1013 else if (cube.get(i) == 0)
1014 {
1015 variables.emplace_back(this->variable_not(i));
1016 }
1017 }
1018 return this->left_fold<ops::AND>(variables);
1019 };
1020
1021 auto const orFold = [this, foldType] (auto& diagrams)
1022 {
1023 switch (foldType)
1024 {
1025 case fold_type::Left:
1026 return this->left_fold<ops::OR>(diagrams);
1027
1028 case fold_type::Tree:
1029 return this->tree_fold<ops::OR>(diagrams);
1030
1031 default:
1032 assert(false);
1033 return this->constant(0);
1034 }
1035 };
1036
1037 auto const& plaLines = file.get_lines();
1038 auto const lineCount = file.get_line_count();
1039 auto const functionCount = file.function_count();
1040
1041 // Create a diagram for each function.
1042 auto functionDiagrams = std::vector<diagram_t>();
1043 functionDiagrams.reserve(functionCount);
1044 for (auto fi = 0; fi < functionCount; ++fi)
1045 {
1046 // First create a diagram for each product.
1047 auto products = std::vector<diagram_t>();
1048 // products.reserve(lineCount);
1049 for (auto li = 0; li < lineCount; ++li)
1050 {
1051 // We are doing SOP so we are only interested
1052 // in functions with value 1.
1053 if (plaLines[as_usize(li)].fVals_.get(fi) == 1)
1054 {
1055 products.emplace_back(product(plaLines[li].cube_));
1056 }
1057 }
1058
1059 // In this case we just have a constant function.
1060 if (products.empty())
1061 {
1062 products.emplace_back(this->constant(0));
1063 }
1064
1065 // Then merge products using OR.
1066 functionDiagrams.emplace_back(orFold(products));
1067 }
1068
1069 return functionDiagrams;
1070}

1073template<expression_node Node>
1074auto diagram_manager<Data, Degree, Domain>::from_expression_tree(
1075 Node const& root
1076) -> diagram_t

1079 unordered_map<std::tuple<node_t*, node_t*>, node_t*, utils::tuple_hash>;
1080
1081 auto constexpr apply_op_wrap = [] (auto const& operation)
1082 {
1083 return [operation] (auto const lhs, auto const rhs)
1084 {
1085 if (lhs == Nondetermined || rhs == Nondetermined)
1086 {
1087 return Nondetermined;
1088 }
1089 return static_cast<int32>(operation(lhs, rhs));
1090 };
1091 };
1092
1093 auto const step
1094 = [this, apply_op_wrap] (auto const& self, auto const& exprNode)
1095 {
1096 if (exprNode.is_constant())
1097 {
1098 return this->constant(exprNode.get_value());
1099 }
1100
1101 if (exprNode.is_variable())
1102 {
1103 return this->variable(exprNode.get_index());
1104 }
1105
1106 assert(exprNode.is_operation());
1107 auto const lhs = self(self, exprNode.get_left());
1108 auto const rhs = self(self, exprNode.get_right());
1109 auto const operation = apply_op_wrap(
1110 [&] (auto const left, auto const right)
1111 {
1112 return exprNode.evaluate(left, right);
1113 }
1114 );
1115
1116 auto applyCache = apply_cache_t();
1117 auto cacheHandle = local_cache_handle<apply_cache_t>(applyCache);
1118
1119 return this->apply_detail(
1120 cacheHandle,
1121 operation,
1122 lhs.unsafe_get_root(),
1123 rhs.unsafe_get_root()
1124 );
1125 };
1126 return step(step, root);

1185template<class Data, degree Degree, domain Domain>
1186template<teddy_bin_op Op, std::random_access_iterator I, std::sentinel_for<I> S>
1187auto diagram_manager<Data, Degree, Domain>::tree_fold(I first, S const last)
1188 -> diagram_t
1189{
1190 static_assert(std::same_as<std::iter_value_t<I>, diagram_t>);
1191
1192 auto const count = std::distance(first, last);
1193 auto currentCount = count;
1194 auto const numOfSteps = static_cast<int64>(std::ceil(std::log2(count)));
1195
1196 for (auto step = 0; step < numOfSteps; ++step)
1197 {
1198 auto const justMoveLast = currentCount & 1;
1199 currentCount = (currentCount / 2) + justMoveLast;
1200 auto const pairCount = currentCount - justMoveLast;
1201
1202 for (auto i = 0; i < pairCount; ++i)
1203 {
1204 *(first + i)
1205 = this->apply<Op>(*(first + 2 * i), *(first + 2 * i + 1));
1206 }
1207
1208 if (justMoveLast)
1209 {
1210 *(first + currentCount - 1)
1211 = std::move(*(first + 2 * (currentCount - 1)));
1212 }
1213 }
1214
1215 return diagram_t(std::move(*first));
1216}

1254 auto data = [] ()
1255 {
1256 if constexpr (CanUseDataMember)
1257 {
1258 // Simply return reference to the data member.
1259 return [] (auto const n) mutable -> decltype(auto)
1260 {
1261 return (n->data());
1262 };
1263 }
1264 else
1265 {
1266 // Return reference to the data that is stored in the map.
1267 return [map = std::unordered_map<node_t*, T>()] (auto const n
1268 ) mutable -> T&
1269 {
1270 // If there is no value for given key [] creates new pair
1271 // and value-initializes the value (0 for primitive types).
1272 return map[n];
1273 };
1274 }
1275 }();
1276
1277 // Actual satisfy count algorithm.
1278 nodes_.traverse_post(
1279 diagram.unsafe_get_root(),
1280 [this, value, &data] (node_t* const node) mutable
1281 {
1282 if (node->is_terminal())
1283 {
1284 data(node) = node->get_value() == value ? 1 : 0;
1285 }
1286 else
1287 {
1288 data(node) = 0;
1289 auto const nLevel = nodes_.get_level(node);
1290 nodes_.for_each_son(
1291 node,
1292 [this, &data, node, nLevel] (auto const son) mutable
1293 {
1294 auto const sonLevel = nodes_.get_level(son);
1295 auto const diff
1296 = nodes_.domain_product(nLevel + 1, sonLevel);
1297 data(node) += data(son) * static_cast<T>(diff);
1298 }
1299 );
1300 }
1301 }
1302 );
1303
1304 auto const rootAlpha = static_cast<int64>(data(diagram.unsafe_get_root()));
1305 auto const rootLevel = nodes_.get_level(diagram.unsafe_get_root());
1306 return rootAlpha * nodes_.domain_product(0, rootLevel);
1307}

1433 self(self, currentLevel + 1, son);
1434 ++sonOrder;

1445 diagram_t const& diagram,
1446 int32 const varIndex,
1447 int32 const varValue
1448) -> diagram_t
1449{

1453 }

1478 varIndex,

1496 };

1646auto diagram_manager<Data, Degree, Domain>::transform_terminal(

1649) -> node_t*
1650{

1652 auto const step
1653 = [this, transformer, &memo] (auto const& self, auto const n)
1654 {

1685template<class Data, degree Degree, domain Domain>
1686template<class Op, class Cache, class... Nodes>
1687auto diagram_manager<Data, Degree, Domain>::apply_detail(
1688 Cache& cache,
1689 Op operation,
1690 Nodes... roots
1691) -> diagram_t
1692{
1693 auto const get_next =
1694 [this] (int32 const minLevel, int32 const sonOrder, node_t* const node)
1695 {
1696 return nodes_.get_level(node) == minLevel ? node->get_son(sonOrder)
1697 : node;
1698 };
1699
1700 auto const get_node_value = [] (node_t* const node)
1701 {
1702 return node->is_terminal() ? node->get_value() : Nondetermined;
1703 };
1704
1705 auto const step = [this, &cache, operation, get_next, get_node_value] (
1706 auto const& self,
1707 auto const... nodes
1708 )
1709 {
1710 auto const cached = cache.lookup(nodes...);
1711 if (cached)
1712 {
1713 return cached;
1714 }
1715
1716 auto const opVal = operation(get_node_value(nodes)...);
1717 auto result = static_cast<node_t*>(nullptr);
1718
1719 if (opVal != Nondetermined)
1720 {
1721 result = nodes_.make_terminal_node(opVal);
1722 }
1723 else
1724 {
1725 auto const minLevel = std::min({nodes_.get_level(nodes)...});
1726 auto const topIndex = nodes_.get_index(minLevel);
1727 auto sons = nodes_.make_sons(
1728 topIndex,
1729 [=] (int32 const sonOrder)
1730 {
1731 return self(self, get_next(minLevel, sonOrder, nodes)...);
1732 }
1733 );
1734
1735 result = nodes_.make_internal_node(topIndex, std::move(sons));
1736 }
1737
1738 cache.put(result, nodes...);
1739 return result;
1740 };
1741
1742 auto const newRoot = step(step, roots...);
1743 auto const newDiagram = diagram_t(newRoot);
1744 nodes_.run_deferred();
1745 return newDiagram;
1746}
