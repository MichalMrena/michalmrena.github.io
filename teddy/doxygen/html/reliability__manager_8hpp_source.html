482 template<component_probabilities Ps>
483 auto reliability_manager<Degree, Domain>::calculate_probabilities
484 (Ps const& ps, diagram_t& sf) -> void
485 {
486 auto const root = sf.get_root();
487
488 this->nodes_.traverse_pre(root, [](auto const n)
489 {
490 n->data() = 0.0;
491 });
492 this->nodes_.for_each_terminal_node([](auto const n)
493 {
494 n->data() = 0.0;
495 });
496 root->data() = 1.0;
497
498 this->nodes_.traverse_level(root, [this, &ps](auto const node)
499 {
500 if (node->is_internal())
501 {
502 auto const nodeIndex = node->get_index();
503 auto k = 0u;
504 this->nodes_.for_each_son(node, [node, nodeIndex, &ps, &k]
505 (auto const son)
506 {
507 son->data() += node->data() * ps[nodeIndex][k];
508 ++k;
509 });
510 }
511 });
512 }

730 template<degree Degree, domain Domain>
731 template<out_var_values Vars, std::output_iterator<Vars> Out>
732 auto reliability_manager<Degree, Domain>::mcvs_g
733 (diagram_t sf, uint_t const j, Out out) -> void
734 {
735 auto const varCount = this->get_var_count();
736 auto dpbdes = std::vector<diagram_t>();
737
738 for (auto i = 0u; i < varCount; ++i)
739 {
740 auto const varDomain = this->nodes_.get_domain(i);
741 for (auto varFrom = 0u; varFrom < varDomain - 1; ++varFrom)
742 {
743 auto const varChange = value_change {varFrom, varFrom + 1};
744 auto const dpbd
745 = this->idpbd_type_3_increase(varChange, j, sf, i);
746 dpbdes.emplace_back(this->to_dpbd_e(varFrom, i, dpbd));
747 }
748 }
749
750 auto const conj = this->template tree_fold<ops::PI_CONJ>(dpbdes);
751 this->template satisfy_all_g<Vars, Out>(1, conj, out);
752 }

754 template<degree Degree, domain Domain>
755 auto reliability_manager<Degree, Domain>::to_dpbd_e
756 ( uint_t varFrom
757 , index_t i
758 , diagram_t dpbd ) -> diagram_t
759 {
760 auto const root = dpbd.get_root();
761 auto const rootLevel = this->nodes_.get_level(root);
762 auto const varLevel = this->nodes_.get_level(i);
763
764 if (varLevel < rootLevel)
765 {
766 auto sons = this->nodes_.make_sons(i,
767 [this, varFrom, root](auto const k)
768 {
769 return k == varFrom
770 ? root
771 : this->nodes_.terminal_node(Undefined);
772 });
773 auto const newRoot = this->nodes_.internal_node(i, std::move(sons));
774 return diagram_t(newRoot);
775 }
776 else
777 {
778 auto memo = std::unordered_map<node_t*, node_t*>();
779 auto const go = [=, this, &memo](auto&& self, auto const n)
780 {
781 if (n->is_terminal())
782 {
783 return n;
784 }
785
786 auto const memoIt = memo.find(n);
787 if (memoIt != std::end(memo))
788 {
789 return memoIt->second;
790 }
791
792 auto const nodeLevel = this->nodes_.get_level(n);
793 auto const nodeIndex = n->get_index();
794 auto sons = this->nodes_.make_sons(nodeIndex,
795 [=, this, &self](auto const k)
796 {
797 auto const son = n->get_son(k);
798 auto const sonLevel = this->nodes_.get_level(son);
799 if (varLevel > nodeLevel && varLevel < sonLevel)
800 {
801 // A new node goes in between the current node
802 // and its k th son.
803 // Transformation does not need to continue.
804 auto newNodeSons = this->nodes_.make_sons(i,
805 [this, varFrom, son](auto const l)
806 {
807 return l == varFrom
808 ? son
809 : this->nodes_.terminal_node(Undefined);
810 });
811 return this->nodes_.internal_node(
812 i, std::move(newNodeSons));
813 }
814 else
815 {
816 // A new node will be inserted somewhere deeper.
817 return self(self, son);
818 }
819 });
820 auto const transformedNode
821 = this->nodes_.internal_node(nodeIndex, std::move(sons));
822 memo.emplace(n, transformedNode);
823 return transformedNode;
824 };
825 return diagram_t(go(go, root));
826 }
827 }

842 template<degree Degree, domain Domain>
843 template<f_val_change F>
844 auto reliability_manager<Degree, Domain>::apply_dpbd
845 (diagram_t lhs, diagram_t rhs, F change) -> diagram_t
846 {
847 using cache_pair = struct { node_t* left; node_t* right; };
848 auto constexpr cache_pair_hash = [](auto const p)
849 {
850 auto const hash1 = std::hash<node_t*>()(p.left);
851 auto const hash2 = std::hash<node_t*>()(p.right);
852 auto result = 0ul;
853 result ^= hash1 + 0x9e3779b9 + (result << 6) + (result >> 2);
854 result ^= hash2 + 0x9e3779b9 + (result << 6) + (result >> 2);
855 return result;
856 };
857 auto constexpr cache_pair_equals = [](auto const l, auto const r)
858 {
859 return l.left == r.left && l.right == r.right;
860 };
861 auto cache = std::unordered_map< cache_pair
862 , node_t*
863 , decltype(cache_pair_hash)
864 , decltype(cache_pair_equals) >();
865
866 auto const go = [this, &cache, change]( auto&& self
867 , auto const l
868 , auto const r ) -> node_t*
869 {
870 auto const cached = cache.find(cache_pair {l, r});
871 if (cached != std::end(cache))
872 {
873 return cached->second;
874 }
875
876 auto const lhsVal = node_value(l);
877 auto const rhsVal = node_value(r);
878 auto const opVal // TODO toto by sa dalo lepsie
879 = lhsVal == Nondetermined || rhsVal == Nondetermined
880 ? Nondetermined
881 : static_cast<uint_t>(change(lhsVal, rhsVal));
882 auto u = static_cast<node_t*>(nullptr);
883
884 if (opVal != Nondetermined)
885 {
886 u = this->nodes_.terminal_node(opVal);
887 }
888 else
889 {
890 auto const lhsLevel = this->nodes_.get_level(l);
891 auto const rhsLevel = this->nodes_.get_level(r);
892 auto const topLevel = std::min(lhsLevel, rhsLevel);
893 auto const topNode = topLevel == lhsLevel ? l : r;
894 auto const topIndex = topNode->get_index();
895 auto sons
896 = this->nodes_.make_sons(topIndex, [=, &self](auto const k)
897 {
898 auto const fst = lhsLevel == topLevel ? l->get_son(k) : l; // TODO tu by bol get_son, ktory by preskakoval fixovane premenne
899 auto const snd = rhsLevel == topLevel ? r->get_son(k) : r;
900 return self(self, fst, snd);
901 });
902
903 u = this->nodes_.internal_node(topIndex, std::move(sons));
904 }
905
906 // TODO in place
907 cache.emplace(std::make_pair(cache_pair {l, r}, u));
908 return u;
909 };
910
911 auto const newRoot = go(go, lhs.get_root(), rhs.get_root());
912 return diagram_t(newRoot);
913 }
