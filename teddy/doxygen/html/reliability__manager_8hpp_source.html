499template<degree Degree, domain Domain>
500template<component_probabilities Ps>
501auto reliability_manager<Degree, Domain>::calculate_probabilities(
502 Ps const& probs,
503 diagram_t const& diagram
504) -> void
505{
506 auto const root = diagram.unsafe_get_root();
507
508 this->nodes_.traverse_pre(
509 root,
510 [] (auto const n)
511 {
512 n->data() = 0.0;
513 }
514 );
515 this->nodes_.for_each_terminal_node(
516 [] (auto const n)
517 {
518 n->data() = 0.0;
519 }
520 );
521 root->data() = 1.0;
522
523 this->nodes_.traverse_level(
524 root,
525 [this, &probs] (auto const node)
526 {
527 if (node->is_internal())
528 {
529 auto const nodeIndex = node->get_index();
530 auto sonOrder = 0;
531 this->nodes_.for_each_son(
532 node,
533 [node, nodeIndex, &probs, &sonOrder] (auto const son)
534 {
535 son->data()
536 += node->data()
537 * probs[as_uindex(nodeIndex)][as_uindex(sonOrder)];
538 ++sonOrder;
539 }
540 );
541 }
542 }
543 );
544}

704template<degree Degree, domain Domain>
705auto reliability_manager<Degree, Domain>::to_dpld_e(
706 int32 const varFrom,
707 int32 const varIndex,
708 diagram_t const& dpld
709) -> diagram_t
710{
711 auto const root = dpld.unsafe_get_root();
712 auto const rootLevel = this->nodes_.get_level(root);
713 auto const varLevel = this->nodes_.get_level(varIndex);
714
715 if (varLevel < rootLevel)
716 {
717 auto sons = this->nodes_.make_sons(
718 varIndex,
719 [this, varFrom, root] (int32 const sonOrder)
720 {
721 return sonOrder == varFrom
722 ? root
723 : this->nodes_.make_terminal_node(Undefined);
724 }
725 );
726 auto const newRoot
727 = this->nodes_.make_internal_node(varIndex, std::move(sons));
728 return diagram_t(newRoot);
729 }
730
731 auto memo = std::unordered_map<node_t*, node_t*>();
732 auto const step = [=, this, &memo] (auto const& self, node_t* const n)
733 {
734 if (n->is_terminal())
735 {
736 return n;
737 }
738
739 auto const memoIt = memo.find(n);
740 if (memoIt != end(memo))
741 {
742 return memoIt->second;
743 }
744
745 auto const nodeLevel = this->nodes_.get_level(n);
746 auto const nodeIndex = n->get_index();
747 auto sons = this->nodes_.make_sons(
748 nodeIndex,
749 [=, this, &self] (int32 const sonOrder)
750 {
751 auto const son = n->get_son(sonOrder);
752 auto const sonLevel = this->nodes_.get_level(son);
753 if (varLevel > nodeLevel && varLevel < sonLevel)
754 {
755 // A new node goes in between the current node
756 // and its k th son.
757 // Transformation does not need to continue.
758 auto newNodeSons = this->nodes_.make_sons(
759 varIndex,
760 [this, varFrom, son] (int32 const newSonOrder)
761 {
762 return newSonOrder == varFrom
763 ? son
764 : this->nodes_.make_terminal_node(Undefined
765 );
766 }
767 );
768 return this->nodes_.make_internal_node(
769 varIndex,
770 std::move(newNodeSons)
771 );
772 }
773
774 // A new node will be inserted somewhere deeper.
775 return self(self, son);
776 }
777 );
778 auto const transformedNode
779 = this->nodes_.make_internal_node(nodeIndex, std::move(sons));
780 memo.emplace(n, transformedNode);
781 return transformedNode;
782 };
783 return diagram_t(step(step, root));
784}

808template<degree Degree, domain Domain>
809template<component_probabilities Ps>
810auto reliability_manager<Degree, Domain>::fussell_vesely_importance(
811 Ps const& probs,
812 diagram_t const& dpld,
813 double const unavailability,
814 int32 const componentState,
815 int32 const componentIndex
816) -> double
817{
818 auto const mnf = this->to_mnf(dpld);
819 auto const mnfProbability = this->calculate_probability(1, probs, mnf);
820 auto nominator = .0;
821 for (auto lowerState = 0; lowerState < componentState; ++lowerState)
822 {

854template<out_var_values Vars, std::output_iterator<Vars> Out>
855auto reliability_manager<Degree, Domain>::mcvs_g(
856 diagram_t const& diagram,
857 int32 const state,
858 Out out
859) -> void
860{
861 auto const varCount = this->get_var_count();
862 auto dpldes = std::vector<diagram_t>();
863
864 for (auto varIndex = 0; varIndex < varCount; ++varIndex)
865 {
866 auto const varDomain = this->nodes_.get_domain(varIndex);
867 for (auto varFrom = 0; varFrom < varDomain - 1; ++varFrom)
868 {
869 auto const varChange = var_change {varIndex, varFrom, varFrom + 1};
870 auto const dpld
871 = this->dpld(varChange, dpld::type_3_increase(state), diagram);
872 dpldes.emplace_back(this->to_dpld_e(varFrom, varIndex, dpld));
873 }
874 }
875
876 auto const conj = this->template tree_fold<ops::PI_CONJ>(dpldes);
877 this->template satisfy_all_g<Vars, Out>(1, conj, out);

907template<degree Degree, domain Domain>
908template<class F>
909auto reliability_manager<Degree, Domain>::apply_dpld_new(
910 diagram_t const& diagram,
911 var_change varChange,
912 F fChange
913) -> diagram_t
914{
915 auto cache = std::unordered_map<
916 std::tuple<node_t*, node_t*>,
917 node_t*,
918 utils::tuple_hash
919 >();
920
921 auto const get_node_value = [] (node_t* const node)
922 {
923 return node->is_terminal() ? node->get_value() : Nondetermined;
924 };
925
926 auto const get_lhs_son = [varChange](node_t* const node, int32 const sonOrder)
927 {
928 auto const son = node->get_son(sonOrder);
929 return son->is_internal() && son->get_index() == varChange.index_
930 ? son->get_son(varChange.from_)
931 : son;
932 };
933
934 auto const get_rhs_son = [varChange](node_t* const node, int32 const sonOrder)
935 {
936 auto const son = node->get_son(sonOrder);
937 return son->is_internal() && son->get_index() == varChange.index_
938 ? son->get_son(varChange.to_)
939 : son;
940 };
941
942 auto const step = [
943 this,
944 &cache,
945 fChange,
946 get_node_value,
947 get_lhs_son,
948 get_rhs_son
949 ] (
950 auto const& self,
951 node_t* const lhs,
952 node_t* const rhs
953 ) -> node_t*
954 {
955 auto const cached = cache.find(std::make_tuple(lhs, rhs));
956 if (cached != end(cache))
957 {
958 return cached->second;
959 }
960
961 auto const lhsVal = get_node_value(lhs);
962 auto const rhsVal = get_node_value(rhs);
963 auto const opVal = lhsVal == Nondetermined || rhsVal == Nondetermined
964 ? Nondetermined
965 : static_cast<int32>(fChange(lhsVal, rhsVal));
966 auto result = static_cast<node_t*>(nullptr);
967
968 if (opVal != Nondetermined)
969 {
970 result = this->nodes_.make_terminal_node(opVal);
971 }
972 else
973 {
974 auto const lhsLevel = this->nodes_.get_level(lhs);
975 auto const rhsLevel = this->nodes_.get_level(rhs);
976 auto const topLevel = std::min(lhsLevel, rhsLevel);
977 auto const topNode = topLevel == lhsLevel ? lhs : rhs;
978 auto const topIndex = topNode->get_index();
979 auto sons = this->nodes_.make_sons(
980 topIndex,
981 [=, &self] (int32 const sonOrder)
982 {
983 auto const fst = lhsLevel == topLevel
984 ? get_lhs_son(lhs, sonOrder)
985 : lhs;
986
987 auto const snd = rhsLevel == topLevel
988 ? get_rhs_son(rhs, sonOrder)
989 : rhs;
990
991 return self(self, fst, snd);
992 }
993 );
994
995 result = this->nodes_.make_internal_node(topIndex, std::move(sons));
996 }
997
998 cache.emplace(
999 std::piecewise_construct,
1000 std::make_tuple(lhs, rhs),
1001 std::make_tuple(result)
1002 );
1003 return result;
1004 };
1005
1006 auto const oldRoot = diagram.unsafe_get_root();
1007 auto const lhsRoot = oldRoot->get_index() == varChange.index_
1008 ? oldRoot->get_son(varChange.from_)
1009 : oldRoot;
1010 auto const rhsRoot = oldRoot->get_index() == varChange.index_
1011 ? oldRoot->get_son(varChange.to_)
1012 : oldRoot;
1013 auto const newRoot = step(step, lhsRoot, rhsRoot);
1014 return diagram_t(newRoot);
1015}

1017template<degree Degree, domain Domain>
1018template<component_probabilities Ps>
1019auto reliability_manager<Degree, Domain>::calculate_ntp(
1020 std::vector<int32> const& selectedValues,
1021 Ps const& probs,
1022 diagram_t const& diagram
1023) -> double
1024{
1025 this->nodes_.for_each_terminal_node(
1026 [] (auto const n)
1027 {
1028 n->data() = 0.0;
1029 }
1030 );
1031
1032 for (auto const selectedValue : selectedValues)
1033 {
1034 auto const terminalNode = this->nodes_.get_terminal_node(selectedValue);
1035 if (terminalNode)
1036 {
1037 terminalNode->data() = 1.0;
1038 }
1039 }
1040
1041 this->nodes_.traverse_post(
1042 diagram.unsafe_get_root(),
1043 [this, &probs] (node_t* const node) mutable
1044 {
1045 if (not node->is_terminal())
1046 {
1047 node->data() = 0.0;
1048 auto const nodeIndex = node->get_index();
1049 this->nodes_.for_each_son(
1050 node,
1051 [=, sonState = 0, &probs] (node_t* const son) mutable
1052 {
1053 node->data()
1054 += son->data()
1055 * probs[as_uindex(nodeIndex)][as_uindex(sonState)];
1056 ++sonState;
1057 }
1058 );
1059 }
1060 }
1061 );
1062 return diagram.unsafe_get_root()->data();
1063}

1065template<degree Degree, domain Domain>

1068{
1069 auto const step = [this] (auto& self, node_t* const node)
1070 {
