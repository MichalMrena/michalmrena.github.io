277 template<component_probabilities Ps>
278 auto reliability_manager<Degree, Domain>::calculate_probabilities
279 (Ps const& ps, diagram_t& sf) -> void
280 {
281 auto const root = sf.get_root();
282
283 this->nodes_.traverse_pre(root, [](auto const n)
284 {
285 n->data() = 0.0;
286 });
287 this->nodes_.for_each_terminal_node([](auto const n)
288 {
289 n->data() = 0.0;
290 });
291 root->data() = 1.0;
292
293 this->nodes_.traverse_level(root, [this, &ps](auto const node)
294 {
295 if (node->is_internal())
296 {
297 auto const nodeIndex = node->get_index();
298 auto k = 0u;
299 this->nodes_.for_each_son(node, [this, node, nodeIndex, &ps, &k]
300 (auto const son)
301 {
302 son->data() += node->data() * ps[nodeIndex][k];
303 ++k;
304 });
305 }
306 });
307 }

432 auto reliability_manager<Degree, Domain>::dpbd_i_3
433 ( value_change const var
434 , uint_t const f
435 , diagram_t const& sf
436 , index_t const i ) -> diagram_t
437 {
438 auto const fVal = this->constant(f);
439 auto const cofactorFrom = this->cofactor(sf, i, var.from);
440 auto const cofactorTo = this->cofactor(sf, i, var.to);
441 if (var.from < var.to)
442 {
443 return this->template apply<ops::AND>
444 ( this->template apply<ops::LESS>(cofactorFrom, fVal)
445 , this->template apply<ops::GREATER_EQUAL>(cofactorTo, fVal) );
446 }
447 else
448 {
449 return this->template apply<ops::AND>
450 ( this->template apply<ops::GREATER_EQUAL>(cofactorFrom, fVal)
451 , this->template apply<ops::LESS>(cofactorTo, fVal) );
452 }
