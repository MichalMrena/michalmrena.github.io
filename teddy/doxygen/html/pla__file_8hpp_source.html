191 inline auto pla_file::load_file
192 (std::string const& path) -> std::optional<pla_file>
193 {
194 namespace rs = std::ranges;
195
196 // Utils:
197 auto constexpr space = [](auto const c)
198 {
199 return std::isspace(c);
200 };
201
202 auto constexpr words = [space](auto const str)
203 {
204 auto ws = std::vector<std::string>();
205 auto it = rs::begin(str);
206 auto const end = rs::end(str);
207
208 while (it != end)
209 {
210 auto const wordBegin = rs::find_if_not(it, end, space);
211 auto const wordEnd = rs::find_if(wordBegin, end, space);
212 if (wordBegin != wordEnd)
213 {
214 ws.emplace_back(wordBegin, wordEnd);
215 }
216 it = wordEnd;
217 }
218
219 return ws;
220 };
221
222 auto ifst = std::ifstream(path);
223 if (not ifst.is_open())
224 {
225 return std::nullopt;
226 }
227
228 // Read options.
229 auto options = std::unordered_map<std::string, std::string>();
230 auto line = std::string();
231 while (std::getline(ifst, line))
232 {
233 // Skip leading spaces.
234 auto const first = rs::find_if_not(line, space);
235 auto const last = std::end(line);
236 if (first == last)
237 {
238 // Skip empty line.
239 continue;
240 }
241
242 if (*first == '#')
243 {
244 // Skip comment.
245 continue;
246 }
247
248 if (*first != '.')
249 {
250 // Not an option.
251 break;
252 }
253
254 // Split into (key, val) pair on the first space.
255 auto const keyLast = rs::find_if(first, last, space);
256 auto const valFirst = keyLast == last
257 ? last
258 : rs::find_if_not(keyLast + 1, last, space);
259 auto key = std::string(first, keyLast);
260 if (valFirst != last)
261 {
262 auto valLast = last;
263 while (space(*(valLast - 1)))
264 {
265 --valLast;
266 }
267 auto val = std::string(valFirst, valLast);
268 options.emplace(std::move(key), std::move(val));
269 }
270 else
271 {
272 options.emplace(std::move(key), std::string());
273 }
274 }
275
276 // Parse header.
277 auto const optionsEnd = rs::end(options);
278 auto const varCountIt = options.find(".i");
279 auto const fCountIt = options.find(".o");
280 auto const lineCountIt = options.find(".p");
281 if ( varCountIt == optionsEnd
282 or fCountIt == optionsEnd
283 or lineCountIt == optionsEnd )
284 {
285 return std::nullopt;
286 }
287
288 auto const varCount = utils::parse<std::size_t>(varCountIt->second);
289 auto const fCount = utils::parse<std::size_t>(fCountIt->second);
290 auto const lineCount = utils::parse<std::size_t>(lineCountIt->second);
291
292 if (not varCount or not fCount or not lineCount)
293 {
294 return std::nullopt;
295 }
296
297 // Read data.
298 auto lines = std::vector<pla_file::pla_line>();
299 lines.reserve(*lineCount);
300 do
301 {
302 auto const first = rs::find_if_not(line, space);
303 auto const last = rs::end(line);
304 if (first == last)
305 {
306 // Skip empty line.
307 continue;
308 }
309
310 if (*first == '.')
311 {
312 // This can only be the .e line.
313 break;
314 }
315
316 // Split on the first space.
317 auto const varsLast = rs::find_if(first, last, space);
318 if (varsLast == last)
319 {
320 return std::nullopt;
321 }
322 auto const fsFirst = rs::find_if_not(varsLast + 1, last, space);
323 auto const fsLast = rs::find_if(fsFirst, last, space);
324 auto const vars = std::string(first, varsLast);
325 auto const fs = std::string(fsFirst, fsLast);
326
327 if (vars.size() != *varCount or fs.size() != *fCount)
328 {
329 return std::nullopt;
330 }
331
332 // Parse variables into cube.
333 auto variables = cube_t(*varCount);
334 for (auto i = 0ul; i < *varCount; ++i)
335 {
336 switch (vars[i])
337 {
338 case '0': variables.set(i, 0u); break;
339 case '1': variables.set(i, 1u); break;
340 case '~':
341 case '-': variables.set(i, cube_t::Undefined); break;
342 default: return std::nullopt;
343 }
344 }
345
346 // Parse functions into cube.
347 auto functions = cube_t(*fCount);
348 for (auto i = 0u; i < *fCount; ++i)
349 {
350 switch (fs[i])
351 {
352 case '0': functions.set(i, 0u); break;
353 case '1': functions.set(i, 1u); break;
354 case '-':
355 case '~': functions.set(i, cube_t::Undefined); break;
356 default: return std::nullopt;
357 }
358 }
359
360 lines.push_back({std::move(variables), std::move(functions)});
361
362 } while (std::getline(ifst, line));
363
364 // Read labels.
365 auto const inLbIt = options.find(".ilb");
366 auto const ouLbIt = options.find(".ob");
367 auto inputLabels = inLbIt != rs::end(options)
368 ? words(inLbIt->second)
369 : std::vector<std::string>();
370 auto outputLabels = ouLbIt != rs::end(options)
371 ? words(ouLbIt->second)
372 : std::vector<std::string>();
373
374 return pla_file( std::move(lines)
375 , std::move(inputLabels)
376 , std::move(outputLabels) );
377 }
